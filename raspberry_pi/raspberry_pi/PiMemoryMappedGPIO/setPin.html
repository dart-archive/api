void setPin(int pin, bool value)

Set the value of the pin to value. The boolean value true represents high (1) and the boolean value false represents low (0).

Source 

void setPin(int pin, bool value) {
  checkPinRange(pin);
  // GPIO output set and output clear registers each have 1 bits for 32 pins.
  int register = pin ~/ 32;
  int shift = pin % 32;
  if (value) {
    _mem.setUint32(_gpioOutputSetBase + (register << 2), 1 << shift);
  } else {
    _mem.setUint32(_gpioOutputClearBase + (register << 2), 1 << shift);
  }
}
